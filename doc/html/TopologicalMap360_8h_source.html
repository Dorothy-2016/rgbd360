309  void RearrangePartition( std::vector<mrpt::vector_uint> &parts )
310  {
311  std::cout << "RearrangePartition...\n";
312  std::vector<mrpt::vector_uint> parts_swap;
313  parts_swap.resize( parts.size() );
314  for( unsigned int countPart1 = 0; countPart1 < parts_swap.size(); countPart1++ ) // First, we arrange the parts regarding its front element (for efficiency in the next step)
315  {
316  unsigned int smallestKF = 0;
317  for( unsigned int countPart2 = 1; countPart2 < parts.size(); countPart2++ )
318  if( parts[smallestKF].front() > parts[countPart2].front() )
319  smallestKF = countPart2;
320  parts_swap[countPart1] = parts[smallestKF];
321  parts.erase( parts.begin() + smallestKF );
322  }
323  parts = parts_swap;
324  }
