501  void buildSphereCloud_fromImage()
502  {
503  double time_start = pcl::getTime();
504 
505  size_t height_SphereImg = sphereRGB.rows;
506  size_t width_SphereImg = sphereRGB.cols;
507  float angle_pixel(width_SphereImg/(2*PI));
508 
509  sphereCloud->resize(height_SphereImg*width_SphereImg);
510  sphereCloud->height = height_SphereImg;
511  sphereCloud->width = width_SphereImg;
512  sphereCloud->is_dense = false;
513 
514  float angle_px(1/angle_pixel);
515  int row_pixels = 0;
516  float offset_phi = PI*31.5/180; // height_SphereImg((width_SphereImg/2) * 63.0/180),
517  float PI_ = PI; // Does this spped-up the next loop?
518  for(int row_phi=0; row_phi < height_SphereImg; row_phi++, row_pixels += width_SphereImg)
519  {
520  float phi_i = row_phi*angle_px - offset_phi;// + PI/2;
521 
522  float sin_phi = sin(phi_i);
523  float cos_phi = cos(phi_i);
524 
525  // Stitch each image from each sensor
526 // #pragma omp parallel for
527  for(int col_theta=0; col_theta < width_SphereImg; col_theta++)
528  {
529  float theta_i = col_theta*angle_px - PI_; // + PI;
530  float depth = sphereDepth.at<float>(row_phi,col_theta);
531 // if(row_phi >50)
532 // cout << "pos " << row_phi << " " << col_theta << " Phi/Theta " << phi_i << " " << theta_i << " depth " << depth << endl;
533  if(depth != 0)
534  {
535 // cout << " Assign point in the cloud\n";
536  sphereCloud->points[row_pixels+col_theta].x = sin(theta_i) * cos_phi * depth;
537 // sphereCloud->points[row_pixels+col_theta].x = cos(theta_i) * cos_phi * depth;
538  sphereCloud->points[row_pixels+col_theta].y = sin_phi * depth;
539  sphereCloud->points[row_pixels+col_theta].z = cos(theta_i) * cos_phi * depth;
540 // sphereCloud->points[row_pixels+col_theta].z = sin(theta_i) * cos_phi * depth;
541  sphereCloud->points[row_pixels+col_theta].r = sphereRGB.at<cv::Vec3b>(row_phi,col_theta)[2];
542  sphereCloud->points[row_pixels+col_theta].g = sphereRGB.at<cv::Vec3b>(row_phi,col_theta)[1];
543  sphereCloud->points[row_pixels+col_theta].b = sphereRGB.at<cv::Vec3b>(row_phi,col_theta)[0];
544  }
545  else
546  {
547  sphereCloud->points[row_pixels+col_theta].x = std::numeric_limits<float>::quiet_NaN ();
548  sphereCloud->points[row_pixels+col_theta].y = std::numeric_limits<float>::quiet_NaN ();
549  sphereCloud->points[row_pixels+col_theta].z = std::numeric_limits<float>::quiet_NaN ();
550  }
551  }
552  }

597  void mergePlanes()
598  {
599  double time_start = pcl::getTime();
600 
601  // Merge repeated planes
602  for(size_t j = 0; j < planes.vPlanes.size(); j++) // numPrevPlanes
603  if(planes.vPlanes[j].curvature < max_curvature_plane)
604  for(size_t k = j+1; k < planes.vPlanes.size(); k++) // numPrevPlanes
605  if(planes.vPlanes[k].curvature < max_curvature_plane)
606  {
607  bool bSamePlane = false;
608 // Eigen::Vector3f center_diff = planes.vPlanes[k].v3center - planes.vPlanes[j].v3center;
609  Eigen::Vector3f close_points_diff;
610  float dist, prev_dist = 1;
611  if( planes.vPlanes[j].v3normal.dot(planes.vPlanes[k].v3normal) > 0.99 )
612  if( fabs(planes.vPlanes[j].d - planes.vPlanes[k].d) < 0.45 )
613 // if( BhattacharyyaDist_(plane1.hist_H, plane2.hist_H) > configLocaliser.hue_threshold )
614 // if( fabs(planes.vPlanes[j].v3normal.dot(center_diff)) < std::max(0.07, 0.03*center_diff.norm() ) )
615  {
616  // Checking distances:
617  // a) Between an vertex and a vertex
618  // b) Between an edge and a vertex
619  // c) Between two edges (imagine two polygons on perpendicular planes)
620  for(unsigned i=1; i < planes.vPlanes[j].polygonContourPtr->size() && !bSamePlane; i++)
621  for(unsigned ii=1; ii < planes.vPlanes[k].polygonContourPtr->size(); ii++)
622  {
623  close_points_diff = mrpt::pbmap::diffPoints(planes.vPlanes[j].polygonContourPtr->points[i], planes.vPlanes[k].polygonContourPtr->points[ii]);
624  dist = close_points_diff.norm();
