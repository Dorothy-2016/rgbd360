66  void loadConstructionSpecs()
67  {
68  Rt_specs[0] = Eigen::Matrix4f::Identity();
69  Rt_specs[0](2,3) = 0.055; // This is the theoretical distance from the first Asus sensor to the center
70 
71  // The pose of each sensor is given by a turn of 45deg around the vertical axis which passes through the device' center
72  Eigen::Matrix4f turn45deg = Eigen::Matrix4f::Identity();
73  turn45deg(1,1) = turn45deg(2,2) = cos(45*PI/180);
74  turn45deg(1,2) = -sin(45*PI/180);
75  turn45deg(2,1) = -turn45deg(1,2);
76  for(unsigned sensor_id = 1; sensor_id < NUM_ASUS_SENSORS; sensor_id++)
77  Rt_specs[sensor_id] = turn45deg * Rt_specs[sensor_id-1];
78  }

186  void CalibrateRotation()
187  {
188  cout << "Calibrate...\n";
189  Eigen::Matrix<float,21,21> hessian; // Hessian of the rotation of the decoupled system
190  Eigen::Matrix<float,21,1> gradient; // Gradient of the rotation of the decoupled system
191  Eigen::Matrix<float,21,1> update_vector;
192  Eigen::Matrix3f jacobian_rot_i, jacobian_rot_ii; // Jacobians of the rotation
193  float accum_error2;
194 
195  // Load the extrinsic calibration from the device' specifications
196  for(unsigned sensor_id = 0; sensor_id < NUM_ASUS_SENSORS; sensor_id++)
197  Rt_estimated[sensor_id] = Rt_specs[sensor_id];
198 
199  Eigen::Matrix4f Rt_estimatedTemp[NUM_ASUS_SENSORS];
200  Rt_estimatedTemp[0] = Rt_estimated[0];
201 
202  // Parameters of the Least-Squares optimization
203  unsigned _max_iterations = 10;
204  float _epsilon_transf = 0.0001;
205  float _convergence_error = 0.000001;
206 
207  float increment = 1000, diff_error = 1000;
208  int it = 0;
209  while(it < _max_iterations && increment > _epsilon_transf && diff_error > _convergence_error)
210  {
211  // Calculate the hessian and the gradient
212  hessian = Eigen::Matrix<float,21,21>::Zero();
213  gradient = Eigen::Matrix<float,21,1>::Zero();
214  accum_error2 = 0.0;
215 
216  // For the fixed camera:
217  unsigned sensor_id = 0;
218  assert( mmCorrespondences[0].count(1) && mmCorrespondences[0][1].getRowCount() >= 3 );
219  assert( mmCorrespondences[0].count(1) && mmCorrespondences[0][1].getRowCount() >= 3 ); // Loop closure condition
220  for(std::map<unsigned, mrpt::math::CMatrixDouble>::iterator it_pair=mmCorrespondences[0].begin();
221  it_pair != mmCorrespondences[0].end(); it_pair++)
222  {
223  unsigned id_corresp2 = 3*(it_pair->first - 1);
224 
225  for(unsigned i=0; i < it_pair->second.getRowCount(); i++)
226  {
227 // float weight = (inliers / it_pair->second(i,3)) / it_pair->second.getRowCount()
228  Eigen::Vector3f n_obs_i; n_obs_i << it_pair->second(i,0), it_pair->second(i,1), it_pair->second(i,2);
229  Eigen::Vector3f n_obs_ii; n_obs_ii << it_pair->second(i,4), it_pair->second(i,5), it_pair->second(i,6);
230  Eigen::Vector3f n_i = Rt_estimated[sensor_id].block(0,0,3,3) * n_obs_i;
231  Eigen::Vector3f n_ii = Rt_estimated[sensor_id+1].block(0,0,3,3) * n_obs_ii;
232  jacobian_rot_ii = skew(n_ii);
233  Eigen::Vector3f rot_error = (n_i - n_ii);
234 // Eigen::Vector3f rot_error = n_i.cross(n_ii);
235  // cout << "rotation error1 " << rot_error.transpose() << endl;
236  accum_error2 += rot_error.dot(rot_error);
237  hessian.block(id_corresp2,id_corresp2,3,3) += jacobian_rot_ii.transpose() * jacobian_rot_ii;
238  gradient.block(id_corresp2,0,3,1) += jacobian_rot_ii.transpose() * rot_error;
239  }
240  }
241 
242  // For the cameras from 1 to 6
243  for(sensor_id = 1; sensor_id < NUM_ASUS_SENSORS-1; sensor_id++)
244  {
245  assert( mmCorrespondences[sensor_id].count(sensor_id+1) && mmCorrespondences[sensor_id][sensor_id+1].getRowCount() >= 3 );
246 // cout << "sensor_id " << sensor_id << endl;
247 // cout << "Rt+1 " << sensor_id << endl;
248 // cout << it_pair->second << endl;
249  for(std::map<unsigned, mrpt::math::CMatrixDouble>::iterator it_pair=mmCorrespondences[sensor_id].begin();
250  it_pair != mmCorrespondences[sensor_id].end(); it_pair++)
251  {
252  unsigned id_corresp1 = 3*(sensor_id-1);
253  unsigned id_corresp2 = 3*(it_pair->first - 1);
254 
255  for(unsigned i=0; i < it_pair->second.getRowCount(); i++)
256  {
257 // float weight = (inliers / it_pair->second(i,3)) / it_pair->second.getRowCount()
258  Eigen::Vector3f n_obs_i; n_obs_i << it_pair->second(i,0), it_pair->second(i,1), it_pair->second(i,2);
259  Eigen::Vector3f n_obs_ii; n_obs_ii << it_pair->second(i,4), it_pair->second(i,5), it_pair->second(i,6);
260  Eigen::Vector3f n_i = Rt_estimated[sensor_id].block(0,0,3,3) * n_obs_i;
261  Eigen::Vector3f n_ii = Rt_estimated[sensor_id+1].block(0,0,3,3) * n_obs_ii;
262  jacobian_rot_i = skew(-n_i);
263  jacobian_rot_ii = skew(n_ii);
264  Eigen::Vector3f rot_error = (n_i - n_ii);
265  accum_error2 += rot_error.dot(rot_error);
266  // cout << "rotation error_i " << rot_error.transpose() << endl;
267  hessian.block(id_corresp1, id_corresp1, 3, 3) += jacobian_rot_i.transpose() * jacobian_rot_i;
268  gradient.block(id_corresp1,0,3,1) += jacobian_rot_i.transpose() * rot_error;
269  hessian.block(id_corresp2, id_corresp2, 3, 3) += jacobian_rot_ii.transpose() * jacobian_rot_ii;
270  gradient.block(id_corresp2,0,3,1) += jacobian_rot_ii.transpose() * rot_error;
271  // Cross term
272  hessian.block(id_corresp1, id_corresp2, 3, 3) += jacobian_rot_i.transpose() * jacobian_rot_ii;
273  }
274 
275  // Fill the lower left triangle with the corresponding cross terms
276  hessian.block(id_corresp2, id_corresp1, 3, 3) = hessian.block(id_corresp1, id_corresp2, 3, 3).transpose();
277  }
278  }
279 
280  cout << "Error accumulated " << accum_error2 << endl;
281 
282  // Solve the rotation
283  update_vector = -hessian.inverse() * gradient;
284  cout << "update_vector " << update_vector.transpose() << endl;
285 
286  // Update rotation of the poses
287  for(sensor_id = 1; sensor_id < NUM_ASUS_SENSORS; sensor_id++)
288  {
289  mrpt::poses::CPose3D pose;
290  mrpt::math::CArrayNumeric< double, 3 > rot_manifold;
291  rot_manifold[0] = update_vector(3*sensor_id-3,0);
292  rot_manifold[1] = update_vector(3*sensor_id-2,0);
293  rot_manifold[2] = update_vector(3*sensor_id-1,0);
294 // rot_manifold[2] = update_vector(3*sensor_id-3,0) / 10; // Limit the turn around the Z (depth) axis
295  mrpt::math::CMatrixDouble33 update_rot = pose.exp_rotation(rot_manifold);
296  // cout << "update_rot\n" << update_rot << endl;
297  Eigen::Matrix3f update_rot_eig;
298  update_rot_eig << update_rot(0,0), update_rot(0,1), update_rot(0,2),
299  update_rot(1,0), update_rot(1,1), update_rot(1,2),
300  update_rot(2,0), update_rot(2,1), update_rot(2,2);
301  Rt_estimatedTemp[sensor_id] = Rt_estimated[sensor_id];
302  Rt_estimatedTemp[sensor_id].block(0,0,3,3) = update_rot_eig * Rt_estimated[sensor_id].block(0,0,3,3);
303  // cout << "old rotation" << sensor_id << "\n" << Rt_estimated[sensor_id].block(0,0,3,3) << endl;
304  // cout << "new rotation\n" << Rt_estimatedTemp[sensor_id].block(0,0,3,3) << endl;
305  }
306 
307  float new_accum_error2 = calcCorrespRotError(Rt_estimatedTemp);
308 
309 // cout << "New rotation error " << new_accum_error2 << endl;
310  // cout << "Closing loop? \n" << Rt_estimated[0].inverse() * Rt_estimated[7] * Rt_78;
311 
312  // Assign new rotations
313  if(new_accum_error2 < accum_error2)
314  for(sensor_id = 1; sensor_id < NUM_ASUS_SENSORS; sensor_id++)
315  Rt_estimated[sensor_id] = Rt_estimatedTemp[sensor_id];
316 // Rt_estimated[sensor_id].block(0,0,3,3) = Rt_estimatedTemp[sensor_id].block(0,0,3,3);
317 
318  increment = update_vector .dot (update_vector);
319  diff_error = accum_error2 - new_accum_error2;
320  ++it;
321  cout << "Iteration " << it << " increment " << increment << " diff_error " << diff_error << endl;
322  }
323 
324  // Make the X axis of the device (the average X axis of all Asus' rotation matrices) coincide with the vertical axis
325  Eigen::Matrix3f hessian_rot = Eigen::Matrix3f::Zero();
326  Eigen::Vector3f gradient_rot = Eigen::Vector3f::Zero();
327  Eigen::Vector3f X_axis; X_axis << 1, 0, 0;
328  float rotation_error2 = 0;
329  for(unsigned i=0; i < NUM_ASUS_SENSORS; i++)
330  {
331  Eigen::Vector3f X_pose = Rt_estimated[i].block(0,0,3,1);
332  Eigen::Vector3f error = X_axis .cross(X_pose);
333  Eigen::Matrix3f jacobian = -skew(X_axis) * skew(X_pose);
334  hessian_rot += jacobian.transpose() * jacobian;
335  gradient_rot += jacobian.transpose() * error;
336  rotation_error2 += error .dot(error);
337  }
338  Eigen::Vector3f rotation_manifold = - hessian_rot.inverse() * gradient_rot;
339  mrpt::poses::CPose3D pose;
340  mrpt::math::CArrayNumeric< double, 3 > rot_manifold;
341  rot_manifold[0] = 0;
342 // rot_manifold[0] = rotation_manifold(0);
343  rot_manifold[1] = rotation_manifold(1);
344  rot_manifold[2] = rotation_manifold(2);
345  mrpt::math::CMatrixDouble33 update_rot = pose.exp_rotation(rot_manifold);
346  cout << "manifold update " << rot_manifold << "\nupdate_rot\n" << update_rot << endl;
347  Eigen::Matrix3f rotation;
348  rotation << update_rot(0,0), update_rot(0,1), update_rot(0,2),
349  update_rot(1,0), update_rot(1,1), update_rot(1,2),
350  update_rot(2,0), update_rot(2,1), update_rot(2,2);
351
