91  return;
92  }
93 
94  {
95  boost::mutex::scoped_lock updateLock(visualizationMutex);
96 
97  // Render the data
98  viz.removeAllPointClouds();
99  viz.removeAllShapes();
100 
101  if (!viz.updatePointCloud (frame360->sphereCloud, "sphereCloud"))
102  viz.addPointCloud (frame360->sphereCloud, "sphereCloud");
103 
104  if(bShowPlanes)
105  {
106  // Draw planes
107  char name[1024];
108 
109  for(size_t i=0; i < frame360->planes.vPlanes.size(); i++)
110  {
111  mrpt::pbmap::Plane &plane_i = frame360->planes.vPlanes[i];
112  sprintf (name, "normal_%u", static_cast<unsigned>(i));
113  pcl::PointXYZ pt1, pt2; // Begin and end points of normal's arrow for visualization
114  pt1 = pcl::PointXYZ(plane_i.v3center[0], plane_i.v3center[1], plane_i.v3center[2]);
115  pt2 = pcl::PointXYZ(plane_i.v3center[0] + (0.5f * plane_i.v3normal[0]),
116  plane_i.v3center[1] + (0.5f * plane_i.v3normal[1]),
117  plane_i.v3center[2] + (0.5f * plane_i.v3normal[2]));
118  viz.addArrow (pt2, pt1, ared[i%10], agrn[i%10], ablu[i%10], false, name);
119 
120  {
121  sprintf (name, "n%u %s", static_cast<unsigned>(i), plane_i.label.c_str());
122 // sprintf (name, "n%u %.1f %.2f", static_cast<unsigned>(i), plane_i.curvature*1000, plane_i.areaHull);
123  viz.addText3D (name, pt2, 0.1, ared[i%10], agrn[i%10], ablu[i%10], name);
124  }
125 
126  sprintf (name, "approx_plane_%02d", int (i));
127  viz.addPolygon<PointT> (plane_i.polygonContourPtr, 0.5 * red[i%10], 0.5 * grn[i%10], 0.5 * blu[i%10], name);
128 
129  if(bColoredPlanes)
130  {
131  sprintf (name, "plane_%02u", static_cast<unsigned>(i));
132  pcl::visualization::PointCloudColorHandlerCustom <PointT> color (plane_i.planePointCloudPtr, red[i%10], grn[i%10], blu[i%10]);
133  viz.addPointCloud (plane_i.planePointCloudPtr, color, name);
134  viz.setPointCloudRenderingProperties (pcl::visualization::PCL_VISUALIZER_POINT_SIZE, 2, name);
135  }
136 
137  }
