300  void grab(FrameRGBD* framePtr)
301  {
302 // std::cout << "RGBDGrabber_OpenNI2::grab...\n";
303  depth_stream.readFrame(&depth_frame);
304  rgb_stream.readFrame(&color_frame);
305  assert((depth_frame.getWidth() == color_frame.getWidth()) || (depth_frame.getHeight() == color_frame.getHeight()));
306 
307 // // Check exposure
308 // std::cout << "exposure set to " << rgb_stream.getCameraSettings()->getExposure() << std::endl;
309 
310  // Fill the color image
311  const openni::RGB888Pixel* imageBuffer = (const openni::RGB888Pixel*)color_frame.getData();
312  framePtr->getRGBImage().create(color_frame.getHeight(), color_frame.getWidth(), CV_8UC3);
313  memcpy( framePtr->getRGBImage().data, imageBuffer, 3*color_frame.getHeight()*color_frame.getWidth()*sizeof(uint8_t) );
314  cv::cvtColor(framePtr->getRGBImage(),framePtr->getRGBImage(),CV_BGR2RGB); //this will put colors right
315 // std::cout << "RGBDGrabber_OpenNI2::grab - fillColor\n";
316 
317  // Fill the depth image
318  const openni::DepthPixel* depthImgRaw = (openni::DepthPixel*)depth_frame.getData();
319  framePtr->getDepthImage().create(depth_frame.getHeight(), depth_frame.getWidth(), CV_16U);
320  memcpy( framePtr->getDepthImage().data, depthImgRaw, depth_frame.getHeight()*depth_frame.getWidth()*sizeof(uint16_t) );
321 // std::cout << "RGBDGrabber_OpenNI2::grab - fillDepth\n";
322  }
