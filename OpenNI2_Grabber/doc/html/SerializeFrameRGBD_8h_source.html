00047 void getMatrixNumberRepresentationOf_uint64_t(uint64_t number,cv::Mat & matrixNumberRepresentation)
00048 {
00049 
00050     //Determine the number of digits of the number
00051     int num_digits = 0;
00052     uint64_t number_aux = number;
00053     while(number_aux > 0)
00054     {
00055       num_digits++;
00056       number_aux/=10;
00057     }
00058 
00059     //Compute the matrix representation of the number
00060     matrixNumberRepresentation = cv::Mat::zeros(1,num_digits,CV_8U);
00061 
00062     uint64_t remainder = number;
00063     for(int digitIndex=0;digitIndex<num_digits;digitIndex++)
00064     {
00065       if(remainder==0){break;}
00066 
00067       uint8_t greaterDigit;
00068       uint64_t dividend = remainder;
00069       uint64_t divisor = pow(10,num_digits-1-digitIndex);
00070       uint64_t quotient = remainder / divisor;
00071       greaterDigit = quotient;
00072       matrixNumberRepresentation.at<uint8_t>(0,digitIndex)=greaterDigit;
00073       remainder = dividend - divisor * quotient;
00074     }
00075 }

00078 {
00079     int num_digits = matrixNumberRepresentation.cols;
00080 
00081     number=0;
00082     uint64_t power10=1;

00084     {
00085         number += power10 * uint64_t(matrixNumberRepresentation.at<uint8_t>(0,digitIndex));
